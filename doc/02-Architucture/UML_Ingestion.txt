    '──────────────────────────────────────────────────────────
    '  COMMON / SHARED PART

    class DocumentLoader {
        + load_documents(subfolder:str) : List[Tuple[str,str]]
        - root : Path
    }

    class Chunker {
        + split(file_path:str, text:str, chunk_size:int=1200, overlap:int=120) : List[Tuple[str,str]]
    }

    class FileManifest {
        + compute_sha256(file_path:str) : str
        + load_manifest(manifest_path:str) : Dict
        + diff(records_now:List[Record], manifest_prev:Dict) : Tuple[List[Record],List[Record],List[Record]]
        + publish_atomic(manifest_dict:Dict, manifest_path:str) : None
    }

    note right of FileManifest
      Record (TypedDict):
      {
        "path":   str,   ' relative to doc_root
        "sha256": str,
        "mtime":  float,
        "size":   int
      }
    end note

    class IngestionStats {
        + files_scanned             : int
        + to_process                : int
        + unchanged                 : int
        + tombstones                : int

        + chunks_added              : int
        + vectors_upserted          : int

        + dense_vectors_upserted    : int
        + sparse_vectors_upserted   : int

        + deleted_old_versions      : int
        + deleted_tombstones        : int

        + embedded_bytes            : int
        + dense_embedded_bytes      : int
        + sparse_embedded_bytes     : int

        + published_manifest_path   : str
    }

    class IngestionManager {
        + run(
            subfolder:str,
            store:VectorStoreChroma,
            chunker:Chunker,
            embedder:Embedder,
            manifest_path:str,
            sparse_store:VectorStoreSplade=None,
            sparse_embedder:SpladeEmbedder=None,
            chunk_size:int=1200,
            overlap:int=120,
            delete_old_versions:bool=True,
            delete_tombstones:bool=False
          ) : IngestionStats

        - doc_root : Path
        - loader   : DocumentLoader

        - _delete_file_version(...) : int
        - _count_ids(...) : int
    }

    note bottom of IngestionManager
      ONE canonical chunking pass:
      - loader reads source files
      - chunker builds canonical chunks once
      - chunk_id = rel_path::sha256::chunk_idx
      - SAME ids + SAME metadata go to BOTH branches

      Dense and SPLADE are parallel representations
      of the SAME chunk universe.
    end note

    '──────────────────────────────────────────────────────────
    '  DENSE PIPELINE (CURRENT / CHROMA)
    '──────────────────────────────────────────────────────────

    class Embedder {
        + embed(texts:List[str]) : List[List[float]]
        - model  : str
        - client : OpenAI
    }

    class ChromaVectorStoreBase {
        + add(ids:List[str], vectors:List[List[float]], metadatas:List[Dict]=None) : None
        + query(vector:List[float], k:int=10, where:Dict=None) : List[str]
        + delete_where(where:Dict) : None
        + snapshot(timestamp:str=None) : Path
        + collection() : Any

        - persist_path    : Path
        - collection_name : str
        - _client         : chromadb.PersistentClient
        - _col            : chromadb.Collection

        - _pre_add(.)   : tuple
        - _post_add(.)  : None
        - _pre_query(.) : tuple
        - _post_query(ids:List[str], raw_result:Dict) : List[str]
    }

    class VectorStoreChroma extends ChromaVectorStoreBase {
        + make_chunk_id(rel_path:str, sha256:str, chunk_idx:int) : str
        + delete_file_version(rel_path:str, sha256:str) : int
        + name() : str
        + persist_root() : Path
        + count() : int
    }

    note bottom of VectorStoreChroma
      Dense document store:
      - physical root: data/chroma_db/<project>
      - logical collection: docs
      - stores dense embedding vectors + metadata
    end note

    '──────────────────────────────────────────────────────────
    '  SPARSE PIPELINE (NEW / SPLADE)
    '──────────────────────────────────────────────────────────

    class SpladeEmbedder {
        + embed(texts:List[str]) : List[Dict[str,float]]

        - model      : str
        - device     : str
        - tokenizer  : Any
        - encoder    : Any
        - max_terms  : int
    }

    class SpladeVectorStoreBase {
        + add(ids:List[str], vectors:List[Dict[str,float]], metadatas:List[Dict]=None) : None
        + query(vector:Dict[str,float], k:int=10, where:Dict=None) : List[str]
        + delete_where(where:Dict) : None
        + snapshot(timestamp:str=None) : Path
        + index() : Any

        - persist_path : Path
        - index_name   : str
        - _index       : Any
        - _meta_store  : Any

        - _pre_add(.)   : tuple
        - _post_add(.)  : None
        - _pre_query(.) : tuple
        - _post_query(ids:List[str], raw_result:Dict) : List[str]
    }

    class VectorStoreSplade extends SpladeVectorStoreBase {
        + make_chunk_id(rel_path:str, sha256:str, chunk_idx:int) : str
        + delete_file_version(rel_path:str, sha256:str) : int
        + name() : str
        + persist_root() : Path
        + count() : int
    }

    note bottom of VectorStoreSplade
      Sparse document store:
      - physical root: data/splade_db/<project>
      - logical index: docs_sparse
      - stores SPLADE sparse term-weight vectors + metadata

      IMPORTANT:
      chunk IDs are NOT backend-owned.
      They are provided by RAGstream and must stay
      identical to the dense branch.
    end note

    '──────────────────────────────────────────────────────────
    '  FLOWS
    '──────────────────────────────────────────────────────────

    DocumentLoader --> Chunker : raw text

    Chunker --> Embedder       : canonical chunks
    Chunker --> SpladeEmbedder : canonical chunks

    Embedder --> VectorStoreChroma           : dense vectors + metadata
    SpladeEmbedder --> VectorStoreSplade     : sparse vectors + metadata

    IngestionManager --> DocumentLoader
    IngestionManager --> Chunker
    IngestionManager --> FileManifest

    IngestionManager --> Embedder
    IngestionManager --> VectorStoreChroma

    IngestionManager --> SpladeEmbedder
    IngestionManager --> VectorStoreSplade
}
