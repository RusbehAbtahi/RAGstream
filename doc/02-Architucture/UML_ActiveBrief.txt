package "Memory Recording Core" as MRC {

    class MemoryManager {
        + capture_pair(...)
        - update_active_brief(...)
        - pending_topic_buffer
    }

    class MemoryRecord {
        + record_id
        + input_text
        + output_text
        + active_retrieval_brief
        + contributor_ids
    }

    MemoryManager --> MemoryRecord : creates + saves
}

package "ActiveBrief Compression" as ABC {

    class MemorySentenceReducer {
        + embedding_model
        + reduce_with_centroid(...)
        + reduce_with_query(...)
    }

    class MemoryActiveBriefRelevanceGate {
        + activebrief_threshold
        + pending_topic_threshold
        + evaluate(...)
    }

    class MemoryActiveRetrievalBriefBuilder {
        + build_for_record(...)
    }

    MemoryActiveRetrievalBriefBuilder --> MemorySentenceReducer : reduce Q/A
    MemoryActiveRetrievalBriefBuilder --> MemoryActiveBriefRelevanceGate : decide route
}

package "Agent Stack" as AG {

    class AgentFactory {
        + get_agent(...)
    }

    class AgentPrompt {
        + compose(...)
        + parse(...)
    }

    class LLMClient {
        + chat(...)
    }

    class AgentJSON {
        memory_activebrief/*.json
    }

    AgentFactory --> AgentPrompt
    AgentPrompt --> AgentJSON
}

package "Runtime Config" as CFG {

    class RuntimeConfig {
        + embedding_model = text-embedding-3-small
        + activebrief_threshold
        + pending_topic_threshold
        + reducer_limits
        + target_brief_tokens
    }
}

package "Later MemoryMerge Boundary" as MM {

    class MemoryMerge {
        + reduced_episodic_records
        + semantic_memory_chunks
        + activebrief_as_reference
        + final_memory_context
    }
}

'──────────────────────────────────────────────────────────────
' VERTICAL PACKAGE ORDER
'──────────────────────────────────────────────────────────────

MRC -[hidden]down-> ABC
ABC -[hidden]down-> AG
AG -[hidden]down-> CFG
CFG -[hidden]down-> MM

'──────────────────────────────────────────────────────────────
' MAIN FLOW
'──────────────────────────────────────────────────────────────

MemoryManager --> MemoryActiveRetrievalBriefBuilder : after durable save
MemoryManager --> MemoryActiveBriefRelevanceGate : pending buffer
MemoryActiveRetrievalBriefBuilder --> AgentFactory : get init/update agent
MemoryActiveRetrievalBriefBuilder --> LLMClient : only if gate passes
MemoryActiveRetrievalBriefBuilder --> MemoryRecord : write/copy brief
MemoryActiveBriefRelevanceGate --> MemoryManager : update/clear pending topic
RuntimeConfig --> MemorySentenceReducer
RuntimeConfig --> MemoryActiveBriefRelevanceGate
RuntimeConfig --> MemoryActiveRetrievalBriefBuilder
MemoryRecord ..> MemoryMerge : future source

note right of MemoryActiveBriefRelevanceGate
  Routes:
  - update brief
  - skip + copy previous
  - store pending topic
  - confirm topic shift
end note

note right of MemoryActiveRetrievalBriefBuilder
  Init agent only for first record.
  Topic shift uses update agent.
end note

note bottom of MemoryMerge
  Semantic memory stays in MemoryMerge,
  not document A3/A4.
  ActiveBrief is reference only.
end note
