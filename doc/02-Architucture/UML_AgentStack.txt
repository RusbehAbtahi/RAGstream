package "Agent Stack" {

    class AgentFactory {
        + load_config(agent_id:str, version:str="001") : Dict
        + get_agent(agent_id:str, version:str="001") : AgentPrompt
        + clear_cache() : None
        - _cache : Dict[str,AgentPrompt]
        - _root  : Path
    }

    class AgentPrompt {
        + from_config(config:Dict) : AgentPrompt
        + compose(
            input_payload:Dict,
            active_fields:List[str]=None
        ) : tuple[List[Dict], Dict]
        + parse(
            raw_output:Any,
            active_fields:List[str]=None
        ) : Dict
        + model_name : str
        + max_output_tokens : int
        - mode : str
        - static_prompt : Dict
        - dynamic_bindings : List[Dict]
        - decision_targets : List[Dict]
    }

    class LLMClient {
        + chat(
            messages:List[Dict],
            model_name:str,
            temperature:float,
            max_output_tokens:int,
            response_format:Dict=None
        ) : str
        - client : OpenAI
    }
}

' Controller/agents obtain configured prompts from factory
AppController --> AgentFactory : get_agent(agent_id, version)
A2_PromptShaper ..> AgentFactory : get_agent(...)
A3_NLIGate ..> AgentFactory : get_agent(...)
A4_Condenser ..> AgentFactory : get_agent(...)
A5_SchemaEnforcer ..> AgentFactory : get_agent(...)
NLP_Splitter ..> AgentFactory : get_agent(...)

' AgentPrompt lifecycle
AgentFactory --> AgentPrompt : from_config(config)

' Agents use AgentPrompt to compose + parse
A2_PromptShaper ..> AgentPrompt : compose() / parse()
A3_NLIGate ..> AgentPrompt : compose() / parse()
A4_Condenser ..> AgentPrompt : compose() / parse()
A5_SchemaEnforcer ..> AgentPrompt : compose() / parse()
NLP_Splitter ..> AgentPrompt : compose() / parse()

' AgentPrompt uses LLMClient
AgentPrompt --> LLMClient : chat()

' Config wiring
Settings .> LLMClient : model/api config
SimpleLogger .> AgentPrompt : log composed messages
SimpleLogger .> LLMClient : log errors/latency

note right of AgentFactory
  Neutral config loader / resolver:
  - resolves data/agents/{agent_id}/{version}.json
  - may inline catalog-backed decision targets
  - transparent config-level cache only
  - no agent business logic, no UI logic
end note

note right of AgentPrompt
  Neutral prompt engine:
  - built from JSON config
  - current runtime modes: selector / classifier
  - compose() builds SYSTEM + USER + response_format
  - parse() validates structured JSON output
  - no A2/A3-specific business rules inside
end note

note right of LLMClient
  Thin LLM wrapper:
  - knows messages / model_name /
    temperature / max_output_tokens /
    response_format
  - no knowledge of controller,
    retrieval, or agent semantics
end note
